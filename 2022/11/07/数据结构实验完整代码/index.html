#include "iostream"
#include "cstdio"
using namespace std;
typedef struct node{
    int coefficient;//系数
    int index;//指数
    node *next;
}*Node, node;
void InputAndDerivation(Node &H){
    H=new node;
    H->next=NULL;
    int coefficient;
    int index;
    Node q=new node;
    q=H;
    while(1){
        scanf("%d %d", &coefficient, &index);
        Node p=new node;
        coefficient *= index;
        index--;	
        p->coefficient=coefficient;
        p->index=index;
        p->next=q->next;
        q->next=p;
        q=p;
        if(getchar()=='\n')
            break;
    }
}
void Output(Node H){
    if(H->next==NULL)  return;
    Node p=new node;
    p=H->next;
    while(1){
        if(p->coefficient==0){
            if(H->next==p){
                cout<<p->coefficient<<" "<<p->index;
                break;
            }else
                break;	
        }
        cout<<p->coefficient<<" "<<p->index;
        if(p->next==NULL)
            break;	
        if(p->next != NULL)
            cout<<" ";
        p=p->next;
    }
}
int main(){
    Node H;
    InputAndDerivation(H);
    Output(H);
    return 0;	
}

#include "iostream"
#include "cstdio"
using namespace std;

char Firstroot[50];
char Mediumroot[50];

typedef struct Tree{
    char data;
    Tree *lch, *rch;
}Tree, *Bitree;

int Position(int root, int right, int left);

Bitree BuildTree(int root, int right, int left){
    if(left<right)
        return NULL;
    
    Bitree p=new Tree;
    p->data=Firstroot[root];
    
    int pos=Position(root, right, left);
    p->lch=BuildTree(root+1, right, pos-1);
    p->rch=BuildTree(root+pos-right+1, pos+1, left);
    return p;
}

int Input(){
    int n;
    cin>>n;

    for(int i=0; i<n; ++i){
        cin>>Firstroot[i];
    }
    
    for(int i=0; i<n; ++i){
        cin>>Mediumroot[i];
    }
    return n;
}

int Position(int root, int right, int left){

    for(int i=right; i<=left; ++i){
        if(Mediumroot[i]==Firstroot[root])
            return i;
    }
    return 0;
}

int Height(Bitree T){
    if(T==NULL)
        return 0;
    return (Height(T->lch) > Height(T->rch) ? Height(T->lch)+1 : Height(T->rch)+1);
}

int main(){
    int n=Input();
    Bitree T=BuildTree(0, 0, n-1);
    int height=Height(T);
    cout<<height;
    return 0;
}

    for(i=0; i<G.vexnum;++i){
            for(j=0; j<G.vexnum; ++j){
                G.arcs[i][j].adj=0;
                G.arcs[j][i].adj=0;
            }
        }	
    for(k=0; k<G.arcnum; ++k){
            cin>>i>>j;
            G.arcs[i-1][j-1].adj=1;
            G.arcs[j-1][i-1].adj=1;
        }

void BFSTraver(MGraph G, int v){
    int count=0;
    int visited[G.vexnum];
    for(int i=0; i<G.vexnum; ++i){
            visited[i]=0;
        }
    queue<int> Q;
    visited[v]=1;
    Q.push(v);
            while(!Q.empty() && count<=6){
                int num=Q.size();
                int array[num];
                
                vex_six_num[v] += num;
                
                for(int i=0; i<num; ++i){
                    array[i]=Q.front();
                    Q.pop();
                }
                for(int i=0; i<num; ++i){
                    for(int w=0; w<G.vexnum; ++w){
                        if(G.arcs[w][array[i]].adj && !visited[w]){
                            visited[w]=1;
                            Q.push(w);
                        }
                    }	
                }
                ++count;
            }
}
int main(){
    MGraph G;
    Creatgraph(G);
    for(int i=0; i<G.vexnum; ++i){
        vex_six_num[i]=0;
    }
    for(int i=0; i<G.vexnum; ++i){
            BFSTraver(G, i);
    }
    for(int i=0; i<G.vexnum; ++i){
        printf("%d:  %.2f%%\n", i+1, 100*(float)vex_six_num[i]/G.vexnum);
    }
    return 0;
}

#include "iostream"
#include "cstdio"
#include "ctype.h"
#include "cstring"
using namespace std;

/*
类似拉链法
文件链表结点结构
*/
typedef struct File_node{
    int numbering;//文件编号
    File_node *next;
}File_node;

//word链表结点结构
typedef struct  Word_node{
    char word[10];
    Word_node *nextword;
    File_node *file;
}Word_node;

//散列表
typedef struct Position{
    Word_node *next;
}Position;
Position WORD[10000];

//存放每一个文件中单词的个数
int File_word_num[100];

int Make_Hash(char* word){
    unsigned int h=0;
    char temp[11];
    char *tt=&temp[0];
    strcpy(temp, word);
    while(*tt != '\0'){
        h = (h<<5)+(*tt++ -'a');
    }
    return h%10000;
}

void Storage(char *word, int pos, int numbering){
    //先查找,在Position[pos]中查找
    Word_node *m=WORD[pos].next;

    while(m!=NULL){
        if(!strcmp(m->word, word))
            break;
        m=m->nextword;
    }

    if(m!=NULL){
        File_node *p=new File_node;
        p->numbering=numbering;
        p->next=m->file;
        m->file=p;
    }else{
        Word_node *p=new Word_node;
        File_node *q=new File_node;

        strcpy(p->word, word);

        p->nextword=WORD[pos].next;
        WORD[pos].next=p;
        p->file=NULL;

        q->numbering=numbering;
        q->next=p->file;
        p->file=q;
    }
}

void Input(){
    int num;
    cin>>num;//文件总数

    char word[11];//单词最长为10
    int i;//计数，单词长度
    int m=0;//记录文件中单词的个数
    int pos;
    char c;

    for(int j=0; j<num; ++j){
        i=0;
        while(1){  //控制一个文件的输入，以#为结束条件
            c=cin.get();
            while(isalpha(c) != 0 && i<10){  //输入的一个字符是字母
                //统一为小写字母
                if(c<'a')
                    c+=32;
                //放入word[10]中
                word[i]=c;
                ++i;
                c=cin.get();
            }
            if(c=='#'){
                for(int i=10; i>=0; --i){
                    word[i]='\0';
                }
                i = 0;
                break;
            }else if(i>=3 && i<=10){
                ++m;//表明该文件输入了一个合法单词
                i=0;
                //调用Hash函数，存入散列表
                pos=Make_Hash(word);
                Storage(word, pos, j);
                for(int i=10; i>=0; --i){
                    word[i]='\0';
                }
            }else{
                for(int i= 10; i>=0; --i){
                    word[i]='\0';
                }
                i=0;
            }
        }//while end
        //一个文件输入结束后，将m值放入File_word_num中
        File_word_num[j]=m;
        m=0;
    }
}

void FindAndRecording(int first, int second){
    int find_file[2];//查询文件的编号
    int find_result[100];
    int a=0;
    int file_word_num[2]={0, 0};

    for(int i=0; i<100; ++i){
        find_result[i]=0;
    }
    find_file[0]=first;
    find_file[1]=second;
    int common_num=0;
    Word_node *p=new Word_node;

    for(int i=0; i<10000; ++i){
        p=WORD[i].next;
        while(p!=NULL){
            File_node *q=new File_node;
            for(int b=0; b<2; ++b){//对要查询的所有文件
                q=p->file;
                while(q!=NULL){
                    if(q->numbering==find_file[b]){
                        find_result[find_file[b]]=1;//表明该词汇下面有这个文件
                        file_word_num[b]++;
                        break;
                    }
                    q=q->next;
                }//while end
            }

            for(int x=0; x<2; ++x){
                if (find_result[find_file[x]] == 1)//表明当前位置下面的词汇，是要查询文件的公共词汇
                {
                    a++;
                }
            }
            
            for(int i=0; i<100; ++i){
                find_result[i]=0;
            }
            if(a==2){
                common_num++;
            }
            a=0;
            p=p->nextword;
        }//while end

    }//for end

    cout<<common_num<<endl;
    cout<<file_word_num[0]<<endl;
    cout<<file_word_num[1]<<endl;
    //输出相似度
    printf("%.1f%%\n", 100*(float)common_num/(file_word_num[0]+ file_word_num[1]-common_num));
}

            }

}

#include "iostream"
#include "string"
using namespace std;
typedef struct SqList{
    string student[3][106];
}SqList;
class Sort{
    public:
    int Partition(SqList &L, int low, int high, int tag){
        Swap(L, 0, low);
        Swap(L, 105, low);
        while(low<high) {
            while (low < high && L.student[tag - 1][high].compare(L.student[tag - 1][105]) >= 0 ) {
                if(L.student[tag - 1][high].compare(L.student[tag - 1][105]) == 0){
                     if(L.student[0][high].compare(L.student[0][105])<0){
                         break;
                     }
                }
                --high;
            }
            Swap(L, low, high);
            while (low < high && L.student[tag - 1][low].compare(L.student[tag - 1][105]) <= 0 ) {
                if(L.student[tag - 1][low].compare(L.student[tag - 1][105]) == 0){
                    if(L.student[0][low].compare(L.student[0][105])>0){
                        break;
                    }
                }
                ++low;
            }
            Swap(L, high, low);
        }
        Swap(L, low, 0);
        return low;  //返回枢轴位置
    }
    void Swap(SqList &L, int m, int n){
        L.student[0][m].assign(L.student[0][n]);
        L.student[1][m].assign(L.student[1][n]);
        L.student[2][m].assign(L.student[2][n]);
    }
    void QSort(SqList &L, int low, int high, int tag){
        int pivotloc;
        if(low<high){
            pivotloc=Partition(L, low, high, tag);
            QSort(L, low, pivotloc-1, tag);
            QSort(L, pivotloc+1, high, tag);
        }
    }
};
int main(){
    int n,tag;
    cin>>n>>tag;
    SqList L;
    for(int i=1; i<=n; ++i){
        cin>>L.student[0][i]>>L.student[1][i]>>L.student[2][i];
        if(L.student[2][i].compare("100")==0){
            L.student[2][i].assign("999");
        }
    }
    Sort test;

        if(L.student[2][i].compare("999")==0){
            L.student[2][i].assign("100");
        }
        cout<<L.student[0][i]<<"\t"<<L.student[1][i]<<"\t"<<L.student[2][i]<<endl;
    }

}
